def solution(n):
    #2부터 차근차근 하나씩 넣어보며 최소의 자연수를 찾아본다
    for i in range(2,n):
        #나머지가 1이 될려면 n-1를 사용해 나머지 0되는 값을 찾는다
        if (n-1)%i == 0:
            return i
    #런타임이 오래 걸리는 것 존재

#타인)

def solution(n):
    return [x for x in range(1,n+1) if n%x==1][0]
#문제 그대로 나머지 1인 값을 구함
#최소의 수이므로 [0]이 첫번째 값이므로 바로 반환

#내꺼가 2부터 시작한 이유도 1부터 하면 나머지가 0이 모두 되므로 range를 2부터 시작한 것임
#이것 또한 런타인 오래 걸리는 것 존재
